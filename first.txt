function Decrypt-String {
    param(
        [string]$CipherTextBase64,
        [string]$Password
    )

    # 1) force $Password to exactly 32 chars
    $padded = ($Password.PadRight(32)).Substring(0,32)
    # 2) UTF8 bytes â†’ 256-bit key
    $keyBytes = [System.Text.Encoding]::UTF8.GetBytes($padded)
    # 3) zero IV
    $iv = New-Object byte[] 16

    # 4) decrypt
    $cipher = [Convert]::FromBase64String($CipherTextBase64)
    $aes    = [System.Security.Cryptography.Aes]::Create()
    $aes.Mode    = 'CBC'
    $aes.Padding = 'PKCS7'
    $aes.Key     = $keyBytes
    $aes.IV      = $iv

    $decryptor = $aes.CreateDecryptor()
    $ms        = New-Object System.IO.MemoryStream(,$cipher)
    $cs        = New-Object System.Security.Cryptography.CryptoStream($ms, $decryptor, [System.Security.Cryptography.CryptoStreamMode]::Read)
    $sr        = New-Object System.IO.StreamReader($cs)
    $plain     = $sr.ReadToEnd()

    $sr.Close(); $cs.Close(); $ms.Close()
    return $plain
}
